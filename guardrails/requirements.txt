# -----------------------------------------------------------------------------
# BLACK - Code Formatter
# -----------------------------------------------------------------------------
# Black is an opinionated Python code formatter that enforces consistent style.
# It reformats code to follow a strict subset of PEP 8.
#
# Key features:
#   - Deterministic output (same input = same output)
#   - Line length of 88 characters (optimal for diffs)
#   - No configuration needed (opinionated by design)
#
# Usage:
#   black .                    # Format all Python files
#   black --check .            # Check without modifying
#   black --diff .             # Show what would change
#
black>=24.4.0

# -----------------------------------------------------------------------------
# RUFF - Fast Python Linter
# -----------------------------------------------------------------------------
# Ruff is an extremely fast Python linter written in Rust.
# It replaces multiple tools: flake8, isort, pyupgrade, and more.
#
# Key features:
#   - 10-100x faster than traditional linters
#   - Built-in auto-fix for many issues
#   - Replaces flake8, isort, pyupgrade, bandit
#   - Modern Python 3.11+ support
#
# Usage:
#   ruff check .               # Lint all files
#   ruff check . --fix         # Lint and auto-fix
#   ruff format .              # Format code (alternative to black)
#
ruff>=0.4.0

# -----------------------------------------------------------------------------
# MYPY - Static Type Checker
# -----------------------------------------------------------------------------
# MyPy analyzes Python code to find type errors without running it.
# It uses type annotations (PEP 484) to catch bugs before runtime.
#
# Key features:
#   - Catches type-related bugs at development time
#   - Improves code documentation through types
#   - Enables better IDE support (autocomplete, refactoring)
#   - Strict mode for maximum safety
#
# Usage:
#   mypy .                     # Type check all files
#   mypy path/to/file.py       # Type check specific file
#   mypy . --strict            # Strict type checking
#
mypy>=1.10.0
